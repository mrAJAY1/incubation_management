              email: decoded.email,

            });
            sendRefreshCookie(res, newRefreshToken);

              new Date().getTime() + parseInt(process.env.RT_COOKIE_LIFE)
            );
            redis.set(

              JSON.stringify({

                expires: redisMaxAge,
              })
            );

              email: decoded.email,
              name: decoded.name,
            });
